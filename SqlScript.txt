CREATE DATABASE Shop1;
USE Shop1

CREATE TABLE Product (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Name VARCHAR(100) NOT NULL,
Price MONEY NOT NULL
);

CREATE TABLE SalesPoint (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Name VARCHAR(100) NOT NULL,
ProvidedProducts INT NOT NULL
);

CREATE TABLE ProvidedProducts (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
ProductId INT NOT NULL,
ProductQuantity INT NOT NULL
);

CREATE TABLE Buyer (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Name VARCHAR(100) NOT NULL,
SalesIds INT NOT NULL
);

CREATE TABLE Sale (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
Date DATE NOT NULL,
Time TIME NOT NULL,
SalesPointId INT NOT NULL,
BuyerId INT,
SalesData INT NOT NULL
);

CREATE TABLE SalesData (
ID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
ProductId INT NOT NULL,
ProductQuantity INT NOT NULL,
ProductIdAmount INT NOT NULL,
TotalAmount INT NOT NULL
);
